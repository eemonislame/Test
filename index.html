#include<stdio.h>
int fibonacci(int n);

int main() {
    int n, c;

    printf("Enter the number of terms: ");
    scanf("%d", &n);

    printf("Fibonacci series:\n");

    for (c = 0; c < n; c++) {
        printf("%d\t", fibonacci(c));
    }
    printf("\n");

    return 0;
}

int fibonacci(int n) {
    if (n == 0) {
        return 0;
    } else if (n == 1) {
        return 1;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}
